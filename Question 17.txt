//17.Create a simple calculator using switch case.
#include <stdio.h>
int main()
{
  float a,b,c;
  printf("enter the value of a=");
  scanf("%f",&a);
  printf("enter the value of b=");
  scanf("%f",&b);
  char ch;
  printf("enter the operators(+,-,*,/)");
  scanf(" %c",&ch);
  switch(ch)
  {
    case '+':
    {
      c=a+b;
      printf("addition=%f",c);
      break;
    }
      case '-':
    {
      c=a-b;
      printf("addition=%f",c);
      break;
    }
        case '*':
    {
      c=a*b;
      printf("addition=%f",c);
      break;
    }
        case '/':
    {
      if(b!=0){
      c=a/b;
      printf("addition=%f",c);
      }
      else
      printf("inffinite");
      break;
    }
    default:
    break;
  }
    
  return 0;
}
